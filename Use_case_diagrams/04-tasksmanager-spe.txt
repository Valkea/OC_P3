rectangle "Gestionnaire de tâches" as g_taskmanager {
	usecase "Voir les tâches" as a_task
	usecase "Ajouter une tâche" as a_task_add_self

	usecase "Organiser les tâches" as a_task_organize #Yellow
	usecase "Editer une tâche" as a_task_edit #Yellow
	usecase "Toutes les tâches" as a_view_all #Cyan
	usecase "Celles des étudiants suivis" as a_view_students #Yellow
	usecase "Les siennes" as a_view_self #Yellow
	usecase "Voir les détails d'une tâche" as a_view_details #Yellow

	usecase "Sélectionner un tuteur" as a_select_mentor #Cyan
	usecase "Enregistrer les détails d'une tâche" as a_input_details #Yellow
	usecase "Sélectionner un étudiant" as a_select_student #Yellow
}

usecase "S'authentifier" as a_auth #Yellow
a_auth <|. g_taskmanager : "\n<<include>>        \n"

'######

a_task_add_self <|.. a_select_mentor : "\n<<extend>>\nEXT: si acteur == Learn@Home"
a_task_add_self <|.. a_select_student : "<<extend>>\nEXT: si acteur != Etudiant"
a_task_add_self ..|> a_input_details : "\n<<include>>"

'######

a_task ..|> a_view_details : "<<include>>"
a_task ..|> a_view_self : "<<include>>"
a_task <|.. a_view_all : "\n<<extends>>\nEXT: si acteur == Learn@Home"
a_task <|.. a_view_students : "<<extends>>\nEXT: si acteur == Tuteur"
a_task ..|> a_task_edit : "\n<<include>>"
a_task ..|> a_task_organize : "\n<<include>>"


'######

user -- a_task

'######
