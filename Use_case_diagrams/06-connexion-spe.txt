	usecase "S'identifier" as a_auth
	usecase "S'inscrire" as a_signup
	usecase "Récupérer le mot de passe" as a_recover

	usecase "Comme Étudiant" as a_as_student #Yellow
	usecase "Comme Tuteur" as a_as_mentor #Yellow
	usecase "Comme membre Learn@Home" as a_as_owner #Cyan

	usecase "Soumettre les infos de récupération" as a_input_recover #Yellow
	usecase "Soumettre les infos d'inscription" as a_input_signup #Yellow
	usecase "Soumettre les infos d'authentification" as a_input_logpass #Yellow
	usecase "Obtenir un token d'identification" as a_get_token #Yellow
	usecase "Recevoir un email" as a_get_email #Yellow
}
